call = (fn, name, query, data, cb) ->
  unless typeof cb is 'function'
    cb = data
    data = null
  instance name, (err, db) ->
    return cb err if err
    unless db[fn]?
      return cb new Error "The #{db.constructor.name} driver didn't support the #{fn}() method."
    db[fn] query, data, cb
